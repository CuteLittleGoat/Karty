KOD #2 — KOLEJNOŚĆ MIEJSC (button/SB/BB) + przesuwanie buttona

export function nextIndex(i: number, n: number): number {
  return (i + 1) % n;
}

export function prevIndex(i: number, n: number): number {
  return (i - 1 + n) % n;
}

export type TablePositions = {
  button: number;     // index gracza na buttonie
  smallBlind: number; // index SB
  bigBlind: number;   // index BB
};

// Zakładamy, że wszyscy gracze są „na miejscu” (2..10 graczy).
export function computePositions(button: number, n: number): TablePositions {
  const smallBlind = nextIndex(button, n);
  const bigBlind = nextIndex(smallBlind, n);
  return { button, smallBlind, bigBlind };
}

export function moveButton(button: number, n: number): number {
  return nextIndex(button, n);
}
